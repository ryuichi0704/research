\begin{abstract}
How should a soft decision stump select its
splitting feature?
We compare three strategies---random
assignment~(R), unconstrained learned
weights~(U), and entmax-gated learned
weights~(E)---in mean-field ensembles.
First, we show that the class-level
comparison for Model~E is algorithmically
meaningful: after reparameterizing the split
direction directly on the simplex, a
simplex mirror flow converges globally to
the minimum empirical risk over the
simplex-parameterized infinite-width E-class
(Section~\ref{sec:convergence}).
This bypasses the structural zeros of the
original entmax-logit dynamics.
Second, we characterize the best predictor in
each class under three data regimes
defined by the geometry of the target's
active directions
(Section~\ref{sec:class}).
The estimation cost depends only on the
model architecture ($\sqrt{\log p/n}$
for R/E, $\sqrt{p/n}$ for~U), while
the approximation cost depends on
whether the model's admissible split
directions match the target:
E is matched to nonneg.\ oblique
directions (simplex), U to arbitrary
signed directions, and R only to
coordinate axes.
No single model dominates across all regimes.
\end{abstract}

% ============================================================
\section{Introduction}
\label{sec:intro}
% ============================================================

Soft decision trees trained by gradient methods
have gained interest as alternatives to classical
greedy trees~\citep{frosst2017distilling,
hazimeh2020tree,popov2020neural}.
A central design choice is feature selection at
each split: one may learn it via
entmax~\citep{peters2019sparse,blondel2020learning},
use unconstrained weight vectors, or assign features
randomly as in random subspace
methods~\citep{ho1998random}.

We compare three strategies in the simplest
tractable setting---mean-field ensembles of
depth-one stumps---and ask two questions:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Is the E-class comparison
  algorithmically meaningful?}\;
  Can a geometry-aligned mirror flow reach
  the global optimum of the
  simplex-parameterized infinite-width
  E-class?
  \item \emph{How well do the best class-level
  predictors perform?}\;
  Across different data geometries, how do the
  three model classes compare in estimation and
  approximation?
\end{enumerate}
Question~(i) is answered in
Section~\ref{sec:convergence} with minimal
data assumptions (only boundedness of the
training sample).
Question~(ii) is answered in
Section~\ref{sec:class} under the full
data model.

\paragraph{Three models.}
\textbf{R}~(Random): each stump is assigned a
random feature, fixed thereafter.
\textbf{U}~(Unconstrained): the splitting
direction is a learned unit-norm vector
(equivalently a two-layer neuron).
\textbf{E}~(Entmax): the splitting direction
passes through $\entmax_\alpha$, a
sparsity-inducing simplex map.

\section{Models and Mean-Field Dynamics}
\label{sec:setup}
% ============================================================

\subsection{Activation and Entmax}

We use $\sigma_\beta(u) = 1/(1+e^{-\beta u})$
with Lipschitz constant $\beta/4$.

\begin{lemma}[Entmax Jacobian
  {\citep{blondel2020learning}}]
  \label{lem:entmax}
  For $\alpha \in (1,2)$,
  $\entmax_\alpha$ is differentiable and
  1-Lipschitz.
  Writing $\mathbf{p} = \entmax_\alpha(\vz)$,
  $s_j = p_j^{2-\alpha}$,
  $S_s = \sum_{\cS} s_m$:
  (a)~$[J_\alpha(\vz)]_{ij} = 0$ for
  $j \notin \cS(\vz)$.
  (b)~For $i,j \in \cS$:
  $[J_\alpha]_{ij}
  = (s_i\delta_{ij} - s_is_j/S_s)/(\alpha{-}1)$.
  (c)~At $\mathbf{p} = (1/p)\vone$:
  $[J_\alpha]_{ij}
  = \eta_\alpha(\delta_{ij} - 1/p)$
  with
  $\eta_\alpha = p^{\alpha-2}/(\alpha{-}1)$.
\end{lemma}


\subsection{Three Models}

\paragraph{Model~E.}
$T_E(\vx;\vtheta)
= c\,\sigma_\beta(\va^\top\vx - b)$,
$\va = \entmax_\alpha(\vw/\tau)$,
$\vtheta = (\vw,b,c)$.

\paragraph{Model~U.}
$T_U(\vx;\vtheta)
= c\,\sigma_\beta(\vw^\top\vx - b)$,
$\norm{\vw}_2 \le 1$,
$\vtheta = (\vw,b,c)$.

\paragraph{Model~R.}
$T_R(\vx;j,b,c)
= c\,\sigma_\beta(x_j - b)$
with $j \sim \mathrm{Uniform}([p])$ fixed at
initialization.

\medskip\noindent
The $M$-stump ensemble class with
variation-norm constraint:
\begin{equation}
  \mathcal{F}_X^M
  = \Big\{
  f(\vx) = \tfrac{1}{M}\textstyle\sum_{i=1}^M
  c_i\,\sigma_\beta(\cdots)
  : |b_i| \le B,\;
  \tfrac{1}{M}\textstyle\sum_i |c_i|
  \le C_{\mathrm{var}}
  \Big\}.
  \label{eq:class}
\end{equation}


\subsection{Mean-Field Gradient Flow}

If
\(
  \mu_M
  =
  \tfrac{1}{M}\sum_{i=1}^M \delta_{\vtheta_i},
\)
then \(f_{\mu_M}\) reproduces the finite ensemble
in \eqref{eq:class}. This motivates the
mean-field predictor
\[
  f_\mu(\vx)
  = \int T_X(\vx;\vtheta)\,\dmu(\vtheta).
\]
For Models~E and~U, whose trainable coordinates
are continuous, we consider the formal
population Euclidean particle flow
\begin{equation}
  \dot{\vtheta}(t)
  = \E_{\vx,y}\!\big[
  r_{\mu_t}(\vx)\,
  \nabla_{\!\vtheta} T_X(\vx;\vtheta)
  \big],
  \qquad
  r_\mu \coloneqq y - f_\mu.
  \label{eq:ode}
\end{equation}
For Model~R, the feature index \(j\) is a static
label fixed at initialization, and only the
continuous coordinates \((b,c)\) evolve:
\[
  \frac{\dd}{\dd t}(b,c)
  =
  \E_{\vx,y}\!\big[
    r_{\mu_t}(\vx)\,
    \nabla_{(b,c)} T_R(\vx;j,b,c)
  \big].
\]
We use these population flows only as a formal
mean-field motivation and to highlight the
structural zeros of the original entmax-logit
dynamics below; we do not analyze here their
well-posedness or whether the Euclidean dynamics
preserves the bounded parameter domain.

For all three models,
$|T_X(\vx;\vtheta)| \le |c|$
(since $\sigma_\beta \in [0,1]$), so
$T_X$ is bounded whenever $|c|$ is bounded.

\begin{assumption}[Bounded data and parameter domain]
  \label{asm:bounded}
  $\norm{\vx}_\infty \le R$ a.s.,
  $|y| \le B_y$ a.s.;
  The admissible parameter domain satisfies
  $|b| \le B$ and
  $|c| \le C_{\max}$ in all three models, and
  $\norm{\vw}_2 \le 1$ in Model~U.
\end{assumption}

Under Assumption~\ref{asm:bounded},
$T_X$ is bounded
($|T_X| \le C_{\max}$),
Lipschitz in the trainable continuous
coordinates
(for Models~E/U in~\(\vtheta\), and for
Model~R in~\((b,c)\); using
$\sigma_\beta \in C^\infty$ and, for
Model~E, differentiability of
$\entmax_\alpha$ for $\alpha \in (1,2)$),
and the population risk
$R[\mu] = \frac{1}{2}\E[(y-f_\mu)^2]$
is well-defined for probability measures
supported on the admissible domain. These are
properties of the model class, not a claim that
the formal Euclidean flow \eqref{eq:ode}
preserves the bounds.

\section{Simplex Mirror Flow for Model~E}
\label{sec:convergence}
% ============================================================

This section does not analyze the original
population Euclidean particle flow in the
entmax logits. Instead, it proves global
convergence for a different optimization
problem: an empirical mirror flow on the
simplex-parameterized E-class.
This is the statement needed to justify the
class-level comparison for Model~E.

\subsection{Why the Original Entmax Flow is Hard}

\begin{proposition}[Structural gradient zeros
  in Model~E]
  \label{prop:structural-zeros}
  For $\alpha \in (1,2)$, $\tau > 0$:
  at any time~$t$, if
  $j \notin \cS(\vw(t)/\tau)$ then
  $\dot{w}_j(t) = 0$
  (Lemma~\ref{lem:entmax}(a)).
  Model~U has no such structural zeros
  ($\sigma_\beta' > 0$ everywhere).
\end{proposition}

\begin{remark}[Discrete vs.\ continuous dynamics]
  Model~E's entmax creates discrete feature
  transitions: features enter and exit the
  support discontinuously, and while outside,
  receive no gradient signal.
  Model~U's dynamics are continuous:
  all weights evolve smoothly at all times.
  This makes Model~E more sensitive to
  initialization and temperature~$\tau$.
\end{remark}


\subsection{Simplex-Parameterized E-Class}

To bypass the structural zeros, we treat the
split direction \(\va\) itself as the primary
parameter.

\begin{definition}[Simplex stump domain]
  \label{def:E-domain}
  Let
  \[
    \Theta_E
    \coloneqq
    \simplex \times [-B,B] \times [-C_{\max},C_{\max}],
  \]
  and for \(\vtheta=(\va,b,c)\in\Theta_E\),
  define
  \[
    \phi_{\vtheta}(\vx)
    \coloneqq
    c\,\sigma_\beta(\va^\top \vx - b).
  \]
  The corresponding infinite-width class is
  \[
    \mathcal{F}_E^\infty
    \coloneqq
    \Bigl\{
      f_\mu(\vx)
      =
      \int_{\Theta_E}\phi_{\vtheta}(\vx)\,\dmu(\vtheta)
      :
      \mu \in \mathcal{P}(\Theta_E)
    \Bigr\}.
  \]
  We call \(\mathcal{F}_E^\infty\) the
  \emph{simplex-parameterized infinite-width
  E-class}.
\end{definition}

\begin{remark}
  The original entmax stump class is contained in
  this simplex-parameterized class, because
  \(\entmax_\alpha(\vw/\tau)\in\simplex\) for
  every \(\vw\).
\end{remark}

\begin{definition}[Empirical risk and mirror flow]
  \label{def:E-flow}
  Fix a training sample
  \(D_n=\{(\vx_i,y_i)\}_{i=1}^n\)
  satisfying Assumption~\ref{asm:bounded}.
  For \(\mu \in \mathcal{P}(\Theta_E)\), define
  \[
    \hat R_n(\mu)
    \coloneqq
    \frac{1}{2n}
    \sum_{i=1}^n
    \bigl(
      y_i - f_\mu(\vx_i)
    \bigr)^2.
  \]
  Let \(m\) be a reference probability measure on
  \(\Theta_E\) with full support, and let
  \(q_t\) be densities with respect to~\(m\).
  Write
  \[
    G_q(\vtheta)
    \coloneqq
    -\frac{1}{n}
    \sum_{i=1}^n
    \bigl(
      y_i - f_{q\,m}(\vx_i)
    \bigr)\phi_{\vtheta}(\vx_i),
  \]
  and
  \[
    \overline G_t
    \coloneqq
    \int_{\Theta_E}
    G_{q_t}(\vtheta)\,q_t(\vtheta)\,\dd m(\vtheta).
  \]
  The simplex mean-field mirror flow is
  \begin{equation}
    \partial_t q_t(\vtheta)
    =
    -q_t(\vtheta)
    \bigl(
      G_{q_t}(\vtheta)-\overline G_t
    \bigr).
    \label{eq:E-mirror}
  \end{equation}
\end{definition}

\begin{theorem}[Global convergence of simplex mirror flow]
  \label{thm:E-global}
  Under Assumption~\ref{asm:bounded}, fix a
  full-support reference measure \(m\) on
  \(\Theta_E\) and an initial density \(q_0\)
  satisfying
  \[
    0 < q_{\min} \le q_0(\vtheta) \le q_{\max}
    < \infty
    \qquad
    \text{$m$-a.e. on \(\Theta_E\).}
  \]
  Then:
  \begin{enumerate}[label=(\alph*),leftmargin=*]
    \item the mirror flow \eqref{eq:E-mirror}
    has a unique global solution
    \(q_t \in C^1([0,\infty);L^1(m))\);
    \item the empirical risk is monotone:
    \(t \mapsto \hat R_n(q_t\,m)\) is
    nonincreasing;
    \item the empirical risk converges to the
    global optimum of the simplex-parameterized
    infinite-width E-class:
    \[
      \hat R_n(q_t\,m)
      \xrightarrow[t\to\infty]{}
      \inf_{\mu \in \mathcal{P}(\Theta_E)}
      \hat R_n(\mu).
    \]
  \end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
  The map \(\mu \mapsto \hat R_n(\mu)\) is convex
  because \(f_\mu\) is linear in~\(\mu\).
  The flow \eqref{eq:E-mirror} is the
  infinite-dimensional exponentiated-gradient
  flow associated with the KL geometry on
  densities.
  Its key identities are:
  \[
    \frac{\dd}{\dd t}\hat R_n(q_t\,m)
    =
    -\operatorname{Var}_{q_t m}(G_{q_t})
    \le 0
  \]
  and, for every comparison density \(\bar q\),
  \[
    \hat R_n(q_t\,m) - \hat R_n(\bar q\,m)
    \le
    -\frac{\dd}{\dd t}
    \mathrm{KL}(\bar q\,m \,\|\, q_t\,m).
  \]
  Integrating the second inequality yields a
  \(1/t\) comparison bound against any
  absolutely continuous competitor; an
  approximation argument then gives convergence
  to the global infimum over
  \(\mathcal{P}(\Theta_E)\).
  Full details are in
  Appendix~\ref{app:e-mirror}.
\end{proof}

\begin{remark}[Meaning of Theorem~\ref{thm:E-global}]
  The theorem proves global convergence for a
  \emph{geometry-correct} optimization of the
  simplex-parameterized E-class.
  It does not prove the same statement for the
  original Euclidean particle flow in the
  entmax logits~\(\vw\), where
  Proposition~\ref{prop:structural-zeros}
  creates face-crossing difficulties.
  For the class-level comparison, however,
  Theorem~\ref{thm:E-global} is the relevant
  statement: the best predictor in the
  simplex-parameterized infinite-width E-class is
  algorithmically reachable by a natural
  simplex-aligned mirror flow.
\end{remark}

\begin{remark}[Finite convex implementation]
  \label{rem:E-eg}
  Sampling a large finite dictionary
  \(\{\vtheta_m\}_{m=1}^M \subset \Theta_E\)
  and optimizing only the mixture weights by
  exponentiated gradient yields a concrete
  finite-particle surrogate.
  The resulting optimization problem is convex on
  the simplex of mixing weights and enjoys the
  standard
  \(O(\sqrt{\log M / T})\) regret rate.
  The formal statement is given in
  Appendix~\ref{app:e-mirror}.
\end{remark}


% ============================================================
\section{Class-Level Comparison}
\label{sec:class}
% ============================================================

We now characterize the best predictor in each
class.
This section introduces the data assumptions
and shows how the comparison depends on
the \emph{geometry} of the target's active
directions relative to each model's admissible
split directions.


\subsection{Data Model and Geometry Regimes}

The admissible split directions are nested:
\[
  \mathcal{A}_R = \{\ve_1,\dots,\ve_p\}
  \;\subset\;
  \mathcal{A}_E = \simplex
  \;\subset\;
  \mathcal{A}_U = \{\vw : \norm{\vw}_2 \le 1\}.
\]
The performance of each model depends on whether
the target's active directions lie in the model's
admissible set.

\begin{assumption}[General data model]
  \label{asm:data}
  (a)~$\E[\vx] = \mathbf{0}$,
  $\E[\vx\vx^\top] = I_p$,
  $\norm{\vx}_\infty \le R$ a.s.
  (b)~The regression function satisfies
  \[
    y = f^*(\vx) + \varepsilon,
    \qquad
    f^*(\vx) = \sum_{m=1}^r g_m(\va_m^\top\vx)
    + u(\vx),
  \]
  where $r \ll p$,
  $\norm{\va_m}_2 = 1$,
  $\E[\varepsilon \mid \vx] = 0$,
  and $\norm{u}_{L^2} \le \eta$
  (model mismatch).
\end{assumption}

\begin{assumption}[Barron regularity]
  \label{asm:barron}
  Each $g_m$ admits
  $g_m(t) = \int c\,\sigma_\beta(t{-}b)\,
  \dd\rho_m(b,c)$
  with variation norm
  $V_m = \inf_{\rho_m}\int|c|\,\dd\rho_m$.
\end{assumption}

\begin{definition}[Three data regimes]
  \label{def:regimes}
  \leavevmode
  \begin{enumerate}[label=(\roman*),nosep]
    \item \emph{Vertex-compatible}:
    $\va_m = \ve_{j_m}$ for all~$m$
    (coordinate-sparse additive).
    \item \emph{Simplex-compatible non-vertex}:
    $\va_m \in \simplex$ for all~$m$,
    but $\va_m \neq \ve_j$ for some~$m$
    (nonneg.\ oblique, e.g.,
    $g(x_1{+}x_2)$).
    \item \emph{Signed-oblique}:
    some $\va_m \notin \simplex$
    (signed directions, e.g.,
    $g(x_1{-}x_2)$).
  \end{enumerate}
\end{definition}

\begin{remark}
  The vertex-compatible regime recovers
  the standard sparse additive model
  ($u \equiv 0$, $r = k$, $\va_m = \ve_{j_m}$).
  The broader formulation reveals
  what each model is \emph{geometrically
  matched} to.
\end{remark}


\subsection{Estimation (Data-Regime Independent)}

The Rademacher complexity depends only on the
\emph{model's} admissible direction set, not
on the target.

\begin{theorem}[Rademacher complexity]
  \label{thm:rad}
  Under Assumption~\ref{asm:data}(a)
  and~\ref{asm:bounded}:
  \begin{enumerate}[label=(\alph*),leftmargin=*]
    \item E/R:\;
    $\Rad(\mathcal{F}_{E/R}^M)
    = O(C_{\mathrm{var}}\beta R
    \sqrt{\log p/n})$.
    \item U:\;
    $\Rad(\mathcal{F}_U^M)
    = O(C_{\mathrm{var}}\beta R
    \sqrt{p/n})$.
  \end{enumerate}
  (For~R, $p \to |J|$ with $|J| = p$
  w.h.p.\ when $M \ge 2p\log p$.)
\end{theorem}

\begin{proof}
  Contraction ($\sigma_\beta{-}1/2$ is
  $(\beta/4)$-Lipschitz, vanishes at~$0$)
  reduces to the linear Rademacher:
  $R\sqrt{2\log p/n}$ (simplex, E/R)
  or $R\sqrt{p/n}$ ($\ell_2$ ball, U).
\end{proof}

\begin{remark}
  The $\sqrt{\log p}$ vs.\ $\sqrt{p}$ gap
  is an architectural consequence of
  simplex vs.\ $\ell_2$ ball---it holds
  in all three data regimes.
\end{remark}


\subsection{Approximation (Data-Regime Dependent)}

Unlike estimation, the approximation error
depends on the \emph{match} between the model's
admissible directions and the target's active
directions.

We first define the two geometry-mismatch
quantities, then state the approximation
theorem.

\begin{definition}[Geometry mismatch]
  \label{def:mismatch}
  \emph{Simplex distance.}\;
  For a unit direction~$\va$,
  $d_E(\va) = \inf_{\bar\va \in \simplex}
  \norm{\va - \bar\va}_2$.
  %
  \emph{Projection bias.}\;
  The additive-projection error of~$f^*$ is
  \[
  \Gamma_{\mathrm{add}}(f^*)
  = \mathrm{Var}(f^*)
  - \textstyle\sum_{j=1}^p
  \mathrm{Var}(\E[f^* \mid x_j]),
  \]
  i.e., the total interaction variance in
  the ANOVA decomposition under independent
  features.
\end{definition}

\begin{remark}[Quantifying projection bias]
  \label{rem:proj-bias}
  For a single ridge
  $g(\va^\top\vx)$ with centered
  nonlinear~$g$
  and independent Gaussian $x_j$,
  the projection bias has the form
  \[
  \Gamma_{\mathrm{add}}
  = \mathrm{Var}(g)
  \cdot \big(1 - \norm{\va}_4^4\big)
  + \textup{higher-order terms}.
  \]
  The ``spread factor''
  $1 - \norm{\va}_4^4 \in [0,\,1{-}1/p]$
  measures how far $\va$ is from a
  coordinate axis:
  $\norm{\ve_j}_4^4 = 1$ (vertex, bias${}=0$);
  $a_l = 1/\sqrt{s}$ on $s$~coordinates gives
  $1 - 1/s$;
  fully spread gives $1 - 1/p \approx 1$.
  For linear~$g$, $\Gamma_{\mathrm{add}} = 0$
  (additive decomposition is exact).
  The bias is significant only when~$g$ is
  nonlinear \emph{and} $\va$ is oblique.
\end{remark}

\begin{theorem}[Approximation by regime]
  \label{thm:approx}
  Under
  Assumptions~\ref{asm:data}--\ref{asm:barron}
  with $C_{\mathrm{var}} \ge \sum_m V_m$ and
  $V_m = V$:
  \begin{enumerate}[label=(\roman*),leftmargin=*]
    \item \emph{Vertex-compatible}
    ($\va_m = \ve_{j_m}$, $\eta = 0$):\;
    E/U achieve $r^2V^2/M$;
    R achieves $O(rpV^2/M)$
    (w.h.p., random allocation).
    \item \emph{Simplex-compatible}
    ($\va_m \in \simplex$):\;
    E/U achieve $r^2V^2/M + \eta^2$;
    R incurs
    \[
    \Gamma_{\mathrm{add}}(f^*)
    + O(rpV^2/M),
    \]
    where $\Gamma_{\mathrm{add}} \ge 0$
    is the projection bias
    (Definition~\ref{def:mismatch};
    quantified in Remark~\ref{rem:proj-bias}).
    \item \emph{Signed-oblique}
    ($\va_m \notin \simplex$):\;
    U achieves $r^2V^2/M + \eta^2$.
    E incurs a simplex-mismatch bias:
    \[
    \inf_{\mathcal{F}_E^M}
    \norm{f^*{-}f}^2
    \le
    \Big(\frac{\beta}{4}
    \textstyle\sum_m V_m\,d_E(\va_m)\Big)^2
    + \frac{r^2V^2}{M}
    + \eta^2.
    \]
    R incurs
    $\Gamma_{\mathrm{add}}(f^*)
    + O(rpV^2/M)$
    (both biases).
  \end{enumerate}
\end{theorem}

\begin{proof}
  (i)~Barron--Jones
  \citep{barron1993universal,jones1992simple}
  for E/U (active directions in
  $\mathcal{A}_E \cap \mathcal{A}_U$);
  per-feature Barron--Jones + Chernoff
  + Lemma~\ref{lem:cross} for R.
  (ii)~E/U: same Barron--Jones
  (directions in $\mathcal{A}_E$);
  R: decompose error as
  $\norm{f^* - h_{\mathrm{add}}^*}^2
  + \norm{h_{\mathrm{add}}^* - f_R}^2$
  where $h_{\mathrm{add}}^*$ is the best
  additive approximation.
  (iii)~For E: replace each $\va_m$ by
  $\bar\va_m \in \simplex$ with
  $\norm{\va_m - \bar\va_m}_2
  = d_E(\va_m)$.
  The $L^2$ error per ridge:
  $\norm{g_m(\va_m^\top\vx)
  - g_m(\bar\va_m^\top\vx)}_{L^2}
  \le (\beta/4)\,V_m\,d_E(\va_m)$,
  using $\sigma_\beta$-Lipschitz and
  $\norm{(\va_m{-}\bar\va_m)^\top\vx}_{L^2}
  = d_E(\va_m)$
  (since $\E[\vx\vx^\top] = I$).
  Sum over $m$ and add Barron--Jones
  discretization.
\end{proof}

the universal $O(r^2V^2/M)$ discretization.}
\label{tab:approx-regime}
\small
\begin{tabular}{@{}lccc@{}}
  \toprule
  Data regime & R & E & U \\
  \midrule
  Vertex
  & $O(rpV^2/M)$
  & $O(r^2V^2/M)$
  & $O(r^2V^2/M)$ \\[3pt]
  Simplex
  & $\Gamma_{\mathrm{add}}(f^*)
    {+}\,O(\frac{rpV^2}{M})$
  & $O(r^2V^2/M)$
  & $O(r^2V^2/M)$ \\[3pt]
  Signed-obl.
  & $\Gamma_{\mathrm{add}}(f^*)
    {+}\,O(\frac{rpV^2}{M})$
  & $O(\beta^2 r^2 V^2 d_E^2)
    {+}\,O(\frac{r^2V^2}{M})$
  & $O(r^2V^2/M)$ \\
  \bottomrule
\end{tabular}
\end{table}

\begin{remark}[Interpretation of the biases]
  $\Gamma_{\mathrm{add}}$ is irreducible
  (does not vanish as $M \to \infty$):
  coordinate-additive stumps fundamentally
  cannot represent nonlinear oblique ridges.
  The simplex bias
  $O(\beta^2 r^2 V^2 d_E^2)$ is also
  irreducible but $p$-independent;
  it depends only on how far the target
  directions are from the simplex.
  Model~U's approximation is always
  $O(r^2V^2/M + \eta^2)$:
  it pays no geometry-mismatch penalty in any
  regime, at the cost of higher estimation
  ($\sqrt{p/n}$ vs.\ $\sqrt{\log p/n}$).
\end{remark}


\subsection{End-to-End Excess Risk}

\begin{theorem}[ERM excess risk]
  \label{thm:excess}
  Let $\hat{f}_X$ be the ERM over
  $\mathcal{F}_X^M$.
  With probability $\ge 1{-}\delta$:
  \[
  R[\hat{f}_X] - R^*
  \le \underbrace{
  \inf_{\mathcal{F}_X^M}\norm{f^*{-}f}^2
  }_{\textup{approx (regime-dep.)}}
  + \underbrace{
  O((B_y{+}C_{\mathrm{var}})
  \Rad(\mathcal{F}_X^M))
  }_{\textup{estimation (regime-indep.)}}
  + \textup{conc.}
  \]
  For Model~E, Section~\ref{sec:convergence}
  shows that the corresponding
  simplex-parameterized infinite-width class
  optimum is algorithmically reachable by
  simplex mirror flow.
\end{theorem}

\begin{remark}[Each model has a winning regime]
  \label{rem:winning}
  \begin{itemize}[nosep]
    \item \textbf{Vertex-compatible}:
    E wins (best estimation $\sqrt{\log p/n}$
    + best approximation $r^2V^2/M$).
    R is safe but pays $\Theta(p/r)$
    in stump count.
    \item \textbf{Simplex-compatible}:
    E and U are both geometry-matched;
    E has better estimation
    ($\sqrt{\log p}$ vs.\ $\sqrt{p}$);
    R is mismatched.
    \item \textbf{Signed-oblique}:
    U is the only geometry-matched model.
    E pays a simplex-mismatch bias;
    R pays both biases.
    U's estimation cost $\sqrt{p/n}$ is
    the price of this flexibility.
  \end{itemize}
\end{remark}


% ============================================================
\section{Discussion}
\label{sec:discussion}
% ============================================================

\paragraph{Summary.}
We studied three feature-selection strategies
in mean-field soft decision stump ensembles
along two axes.
On the \emph{optimization} axis
(Section~\ref{sec:convergence}):
we showed that Model~E admits a
geometry-correct global optimization theory:
after reparameterizing split directions
directly on the simplex,
the simplex mirror flow converges to the
global optimum of the simplex-parameterized
infinite-width E-class.
On the \emph{performance} axis
(Section~\ref{sec:class}):
the excess risk decomposes into
a regime-independent estimation term
(governed by the simplex-vs-ball
architecture) and a regime-dependent
approximation term (governed by the
geometry match between model and data).
The three data regimes
(vertex-compatible, simplex-compatible,
signed-oblique) each favor a different model
(Table~\ref{tab:approx-regime},
Remark~\ref{rem:winning}).

\paragraph{No single model dominates.}
E has the best estimation
($\sqrt{\log p/n}$) and the best
approximation when the target is
simplex-compatible---but it is
geometry-mismatched for signed-oblique
targets.
U handles all target geometries at the cost
of $\sqrt{p/n}$ estimation.
R avoids all dynamics overhead but is
limited to coordinate-aligned targets and
pays a $\Theta(p/r)$ stump penalty.
The \emph{choice of model is a bet on the
data geometry}: vertex → E or R;
simplex → E; signed-oblique → U.

\paragraph{Limitations.}
Our setting is stylized: depth-one stumps
and uncorrelated features.
The optimization theorem in
Section~\ref{sec:convergence} concerns the
simplex mirror flow for the
simplex-parameterized E-class, not the
original Euclidean particle flow in the
entmax logits.
Whether the original entmax-logit dynamics
also converges globally remains open.
We do not prove comparable global
optimization statements here for Model~U.
Empirical validation is not provided.

\subsection{Proofs for Section~\ref{sec:convergence}}
\label{app:e-mirror}

\begin{proposition}[Convexity of the simplex E-risk]
  \label{prop:E-convex}
  The map
  \(
    \mu \mapsto \hat R_n(\mu)
  \)
  is convex on \(\mathcal{P}(\Theta_E)\).
\end{proposition}

\begin{proof}
  The predictor
  \(
    f_\mu(\vx)
    =
    \int_{\Theta_E}\phi_{\vtheta}(\vx)\,\dmu(\vtheta)
  \)
  is linear in~\(\mu\), and
  \[
    (u_1,\dots,u_n)
    \mapsto
    \frac{1}{2n}\sum_{i=1}^n (y_i-u_i)^2
  \]
  is convex.
  Composing a convex map with a linear map
  gives convexity.
\end{proof}

\begin{proposition}[Functional derivative]
  \label{prop:E-derivative}
  For a density \(q\) with respect to~\(m\),
  let
  \[
    r_q(\vx_i)
    \coloneqq
    y_i - f_{q\,m}(\vx_i).
  \]
  Then the first variation of \(\hat R_n(q\,m)\)
  is
  \[
    G_q(\vtheta)
    =
    -\frac{1}{n}
    \sum_{i=1}^n
    r_q(\vx_i)\,\phi_{\vtheta}(\vx_i).
  \]
  Moreover,
  \begin{align}
    \norm{G_q}_{L^\infty(m)}
    &\le C_{\max}(B_y + C_{\max}),
    \label{eq:app-G-bdd}\\
    \norm{G_q-G_{\tilde q}}_{L^\infty(m)}
    &\le
    C_{\max}^2 \norm{q-\tilde q}_{L^1(m)}.
    \label{eq:app-G-lip}
  \end{align}
\end{proposition}

\begin{proof}
  Let \(h \in L^1(m)\) with \(\int h\,\dd m=0\).
  Since
  \[
    f_{(q+\varepsilon h)m}(\vx_i)
    =
    f_{q\,m}(\vx_i)
    + \varepsilon
    \int \phi_{\vtheta}(\vx_i)h(\vtheta)\,\dd m(\vtheta),
  \]
  differentiating at \(\varepsilon=0\) yields
  \[
    \frac{\dd}{\dd\varepsilon}
    \hat R_n((q+\varepsilon h)m)\Big|_{\varepsilon=0}
    =
    \int G_q(\vtheta)h(\vtheta)\,\dd m(\vtheta).
  \]
  This gives the derivative formula.

  Because \(\abs{\phi_{\vtheta}(\vx_i)} \le \abs{c}
  \le C_{\max}\),
  \[
    \abs{f_{q\,m}(\vx_i)}
    \le
    \int \abs{\phi_{\vtheta}(\vx_i)}q(\vtheta)\,\dd m(\vtheta)
    \le C_{\max},
  \]
  hence
  \(
    \abs{r_q(\vx_i)} \le B_y + C_{\max}
  \),
  which proves \eqref{eq:app-G-bdd}.
  Also,
  \[
    \abs{f_{q\,m}(\vx_i)-f_{\tilde q\,m}(\vx_i)}
    \le
    C_{\max}\norm{q-\tilde q}_{L^1(m)},
  \]
  and substituting this in the formula for
  \(G_q-G_{\tilde q}\) gives
  \eqref{eq:app-G-lip}.
\end{proof}

\begin{proposition}[Well-posedness of the mirror flow]
  \label{prop:E-wellposed}
  Under the assumptions of
  Theorem~\ref{thm:E-global}, the mirror flow
  \eqref{eq:E-mirror} has a unique global
  solution \(q_t \in C^1([0,\infty);L^1(m))\).
  For every \(t\ge 0\):
  \begin{enumerate}[label=(\alph*),nosep]
    \item \(q_t\) is a probability density,
    \item \(q_t(\vtheta) > 0\) for \(m\)-a.e.
    \(\vtheta\),
    \item
    \[
      q_t(\vtheta)
      =
      q_0(\vtheta)\,
      \exp\!\Bigl(
        -\int_0^t
        (G_{q_s}(\vtheta)-\overline G_s)\,\dd s
      \Bigr).
    \]
  \end{enumerate}
\end{proposition}

\begin{proof}
  Define
  \[
    \mathcal V(q)
    \coloneqq
    -q\bigl(G_q-\langle G_q,q\rangle_m\bigr),
    \qquad
    \langle G_q,q\rangle_m
    \coloneqq
    \int G_q q\,\dd m.
  \]
  By Proposition~\ref{prop:E-derivative},
  \(G_q\) is bounded uniformly and Lipschitz from
  \(L^1(m)\) to \(L^\infty(m)\). Thus
  \(\mathcal V\) is locally Lipschitz on
  \(L^1(m)\), so Picard--Lindel\"of gives a
  unique local solution \(q_t\) on a maximal
  interval \([0,T_{\max})\).

  Integrating \eqref{eq:E-mirror} over
  \(\Theta_E\) gives
  \[
    \frac{\dd}{\dd t}\int q_t\,\dd m = 0,
  \]
  hence the total mass stays equal to~\(1\).
  Therefore \(\norm{q_t}_{L^1(m)}=1\) on
  \([0,T_{\max})\). Since
  \(\norm{G_q}_{L^\infty(m)}
  \le C_{\max}(B_y+C_{\max})\),
  \[
    \norm{\mathcal V(q_t)}_{L^1(m)}
    \le
    \norm{q_t}_{L^1(m)}
    \norm{
      G_{q_t}-\langle G_{q_t},q_t\rangle_m
    }_{L^\infty(m)}
    \le 2C_{\max}(B_y+C_{\max}),
  \]
  so blow-up cannot occur in finite time and the
  local solution extends globally.
  The explicit formula follows by solving the
  scalar ODE pointwise in~\(\vtheta\), and the
  positivity of \(q_0\) is preserved.
\end{proof}

\begin{proposition}[Risk dissipation and KL inequality]
  \label{prop:E-kl}
  Along the mirror flow:
  \begin{align}
    \frac{\dd}{\dd t}\hat R_n(q_t\,m)
    &=
    -\operatorname{Var}_{q_t m}(G_{q_t})
    \le 0,
    \label{eq:app-dissipation}\\
    \hat R_n(q_t\,m)-\hat R_n(\bar q\,m)
    &\le
    -\frac{\dd}{\dd t}
    \mathrm{KL}(\bar q\,m \,\|\, q_t\,m)
    \label{eq:app-kl}
  \end{align}
  for every comparison density \(\bar q\) with
  \(\mathrm{KL}(\bar q\,m \,\|\, q_0\,m)<\infty\).
  Consequently, for every \(T>0\),
  \begin{equation}
    \hat R_n(q_T\,m)-\hat R_n(\bar q\,m)
    \le
    \frac{
      \mathrm{KL}(\bar q\,m \,\|\, q_0\,m)
    }{T}.
    \label{eq:app-ergodic}
  \end{equation}
\end{proposition}

\begin{proof}
  By the chain rule and
  Proposition~\ref{prop:E-derivative},
  \[
    \frac{\dd}{\dd t}\hat R_n(q_t\,m)
    =
    \int G_{q_t}(\vtheta)\,
    \partial_t q_t(\vtheta)\,\dd m(\vtheta).
  \]
  Substituting \eqref{eq:E-mirror} gives
  \[
    -\int G_{q_t}^2 q_t\,\dd m
    + \Bigl(\int G_{q_t}q_t\,\dd m\Bigr)^2,
  \]
  which is
  \(-\operatorname{Var}_{q_t m}(G_{q_t})\).

  For the KL derivative, use
  Proposition~\ref{prop:E-wellposed}:
  \[
    \frac{\dd}{\dd t}
    \mathrm{KL}(\bar q\,m \,\|\, q_t\,m)
    =
    -\int \bar q(\vtheta)
    \frac{\partial_t q_t(\vtheta)}{q_t(\vtheta)}
    \,\dd m(\vtheta).
  \]
  Substituting \eqref{eq:E-mirror},
  \[
    \frac{\dd}{\dd t}
    \mathrm{KL}(\bar q\,m \,\|\, q_t\,m)
    =
    \int G_{q_t}(\bar q-q_t)\,\dd m.
  \]
  By convexity of \(\hat R_n\)
  (Proposition~\ref{prop:E-convex}),
  \[
    \hat R_n(\bar q\,m)
    \ge
    \hat R_n(q_t\,m)
    + \int G_{q_t}(\bar q-q_t)\,\dd m,
  \]
  which yields \eqref{eq:app-kl}. Integrating
  over \([0,T]\) and using the monotonicity from
  \eqref{eq:app-dissipation} gives
  \eqref{eq:app-ergodic}.
\end{proof}

\begin{lemma}[Absolutely continuous approximation]
  \label{lem:E-approx}
  Let \(\mu^\star \in \mathcal{P}(\Theta_E)\).
  For every \(\varepsilon>0\), there exists a
  bounded density \(\bar q_\varepsilon\) with
  respect to~\(m\) such that
  \[
    \hat R_n(\bar q_\varepsilon\,m)
    \le
    \hat R_n(\mu^\star) + \varepsilon
  \]
  and
  \(
    \mathrm{KL}(\bar q_\varepsilon\,m \,\|\, q_0\,m)
    < \infty.
  \)
\end{lemma}

\begin{proof}
  Since \(\Theta_E\) is compact and the family of
  functions
  \(
    \{\vtheta \mapsto \phi_{\vtheta}(\vx_i)\}_{i=1}^n
  \)
  is continuous, it is uniformly continuous.
  Hence there exists \(\delta>0\) such that,
  whenever \(d(\vtheta,\vtheta')\le\delta\),
  \[
    \max_{1\le i\le n}
    \abs{
      \phi_{\vtheta}(\vx_i)-\phi_{\vtheta'}(\vx_i)
    }
    \le \eta_\varepsilon
  \]
  for a tolerance \(\eta_\varepsilon>0\) chosen
  sufficiently small below.

  Choose a finite Borel partition
  \(\Theta_E=\bigsqcup_{\ell=1}^L A_\ell\) with
  \(\operatorname{diam}(A_\ell)\le\delta\), and
  pick \(\vtheta_\ell \in A_\ell\). Define
  \[
    \tilde\mu
    \coloneqq
    \sum_{\ell=1}^L
    \mu^\star(A_\ell)\,\delta_{\vtheta_\ell}.
  \]
  Then
  \[
    \max_{1\le i\le n}
    \abs{
      f_{\mu^\star}(\vx_i)-f_{\tilde\mu}(\vx_i)
    }
    \le \eta_\varepsilon.
  \]

  After merging repeated support points, assume
  \(\vtheta_1,\dots,\vtheta_L\) are distinct.
  Choose pairwise disjoint open neighborhoods
  \(U_\ell \ni \vtheta_\ell\) so small that
  \(m(U_\ell)>0\) for every \(\ell\) (by full
  support of~\(m\)) and
  \[
    \sup_{\vtheta \in U_\ell}
    \max_{1\le i\le n}
    \abs{
      \phi_{\vtheta}(\vx_i)-\phi_{\vtheta_\ell}(\vx_i)
    }
    \le \eta_\varepsilon.
  \]
  Define
  \[
    \bar q_\varepsilon(\vtheta)
    \coloneqq
    \sum_{\ell=1}^L
    \mu^\star(A_\ell)\,
    \frac{\mathbf{1}_{U_\ell}(\vtheta)}{m(U_\ell)}.
  \]
  Because the \(U_\ell\) are disjoint,
  \(\bar q_\varepsilon\) is a bounded density.
  Moreover,
  \[
    \max_{1\le i\le n}
    \abs{
      f_{\bar q_\varepsilon m}(\vx_i)-f_{\tilde\mu}(\vx_i)
    }
    \le \eta_\varepsilon.
  \]
  The empirical risk depends only on the finitely
  many bounded moments
  \(
    \int \phi_{\vtheta}(\vx_i)\,\dmu(\vtheta),
  \)
  so choosing \(\eta_\varepsilon\) sufficiently
  small gives
  \[
    \hat R_n(\bar q_\varepsilon\,m)
    \le
    \hat R_n(\mu^\star) + \varepsilon.
  \]
  Since \(q_0 \ge q_{\min}>0\) a.e.,
  \(\mathrm{KL}(\bar q_\varepsilon\,m \,\|\, q_0\,m)\)
  is finite.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:E-global}]
  Existence and monotonicity are exactly
  Proposition~\ref{prop:E-wellposed} and
  \eqref{eq:app-dissipation}.

  Let
  \[
    \hat R_n^\star
    \coloneqq
    \inf_{\mu \in \mathcal{P}(\Theta_E)}
    \hat R_n(\mu),
  \]
  and choose \(\mu^\star\) attaining this
  infimum. Such a minimizer exists because
  \(\Theta_E\) is compact and
  \(\hat R_n\) is continuous under weak
  convergence of measures.

  Fix \(\varepsilon>0\), and let
  \(\bar q_\varepsilon\) be given by
  Lemma~\ref{lem:E-approx}. Applying
  \eqref{eq:app-ergodic} to \(\bar q_\varepsilon\)
  yields
  \[
    \hat R_n(q_T\,m)
    -
    \hat R_n(\bar q_\varepsilon\,m)
    \le
    \frac{
      \mathrm{KL}(\bar q_\varepsilon\,m \,\|\, q_0\,m)
    }{T}.
  \]
  Therefore
  \[
    \limsup_{T\to\infty}
    \bigl(
      \hat R_n(q_T\,m) - \hat R_n^\star
    \bigr)
    \le \varepsilon.
  \]
  Since \(\varepsilon\) is arbitrary and
  \(\hat R_n(q_T\,m)\ge \hat R_n^\star\), the
  limit exists and equals~\(0\).
\end{proof}

\begin{proposition}[Finite convex surrogate]
  \label{prop:E-eg}
  Sample
  \(\vtheta_1,\dots,\vtheta_M \in \Theta_E\)
  from the reference measure~\(m\), and define
  \[
    f_\pi(\vx)
    \coloneqq
    \sum_{m=1}^M \pi_m \phi_{\vtheta_m}(\vx),
    \qquad
    \pi \in \Delta^{M-1}.
  \]
  Then
  \(
    \pi \mapsto \hat R_n(\pi)
  \)
  is convex on \(\Delta^{M-1}\).
  Exponentiated gradient on~\(\pi\) satisfies
  the standard regret bound
  \[
    \frac{1}{T}
    \sum_{t=0}^{T-1}
    \bigl(
      \hat R_n(\pi^t)-\hat R_n(\pi^\star)
    \bigr)
    \le
    O\!\Bigl(
      C_{\max}(B_y+C_{\max})
      \sqrt{\frac{\log M}{T}}
    \Bigr).
  \]
\end{proposition}

\begin{proof}
  Convexity follows because \(f_\pi\) is linear
  in~\(\pi\) and the squared loss is convex.
  The partial derivatives satisfy
  \[
    \Bigl|
    \frac{\partial \hat R_n}{\partial \pi_m}
    \Bigr|
    \le C_{\max}(B_y+C_{\max}),
  \]
  so the standard mirror-descent regret bound on
  the finite simplex applies.
\end{proof}

\subsection{Cross-Term Bound}

\begin{lemma}
  \label{lem:cross}
  For $h_j(x_j)$ of independent variables
  with $|S|{=}k$:
  $\E[(\sum_S h_j)^2]
  \le (1{+}k)\sum_S\E[h_j^2]$.
\end{lemma}

\begin{proof}
  $\E[(\sum h_j)^2]
  = \sum\E[h_j^2]
  + 2\sum_{j<l}\E[h_j]\E[h_l]$.
  By Cauchy--Schwarz:
  $|2\sum_{j<l}a_ja_l|
  \le (k{-}1)\sum a_j^2
  \le k\sum\E[h_j^2]$.
\end{proof}

\title{A Fixed-\texorpdfstring{$K$}{K} Width Mean-Field Limit for Bias-Free BatchEnsemble without Input Modulation}
\author{}
\date{\today}

\begin{document}
\maketitle

\begin{abstract}
We study the fixed-$K$, width-to-infinity limit of a bias-free two-layer BatchEnsemble-style model in the regime without input-side modulation. In this setting all head-specific variables live at the neuron level, so the large-width object is a single deterministic transport PDE on the neuron law rather than a coupled PDE-ODE system. We prove global characteristic well-posedness, stability, and finite-width particle convergence for an abstract bounded-Lipschitz model, and show that standard deterministic-law random initialization yields a deterministic infinite-width predictor vector. A further structural consequence is that permutation-symmetric initial laws remain symmetric under the mean-field flow, forcing exact collapse of all head predictors in the infinite-width limit. The note isolates the cleanest fixed-$K$ width mean-field regime before reintroducing head-global variables such as input-side fast weights.
\end{abstract}

\section{Introduction}

Fix the ensemble size $K$ and send the width $m$ to infinity. If the model contains head-global variables, such as input-side modulations $s_\alpha$, then the natural large-width object is typically a coupled PDE-ODE system driven by a finite-dimensional random environment. But before studying that richer regime, it is useful to isolate the simpler no-input-modulation case in which all head-specific parameters are attached to individual neurons.

The concrete model we analyze is the bias-free two-layer BatchEnsemble-style network
\begin{equation}
  f_\alpha^m(x)
  :=
  \frac1m\sum_{j=1}^m
  a_j\,\sigma\!\bigl(r_{\alpha,j}\,w_j^\top x\bigr),
  \qquad
  \alpha\in\{1,\dots,K\},
  \label{eq:intro-model}
\end{equation}
where $w_j\in\R^{d_x}$ and $a_j\in\R$ are shared across heads while
$r_{\alpha,j}\in\R$ is a head-specific modulation attached to neuron $j$. The key
feature is that the entire head dependence is already contained in the single-neuron
state
\[
  \theta_j=(a_j,w_j,r_{1,j},\dots,r_{K,j}).
\]
As a result, the width limit is a standard single-population mean-field transport
equation on the law of $\theta$.

This changes the message relative to the setting with head-global variables.
For the model \eqref{eq:intro-model}, standard i.i.d.\ initialization from a
deterministic law produces a deterministic infinite-width limit: width averages
out all order-one randomness. Moreover, if the initialization law is invariant
under permutations of the head coordinates $(r_1,\dots,r_K)$, then the mean-field
flow preserves that symmetry, and the limiting head predictors collapse exactly:
\[
  f_1(\cdot;\rho_t)=\cdots=f_K(\cdot;\rho_t).
\]
Thus in this no-input-modulation regime, leading-order head diversity requires an
asymmetric initial law, a finite-width fluctuation analysis, or a different
asymptotic question such as $K\to\infty$.

\section{Abstract No-Input Framework}

\subsection{State variables and outputs}

Fix an ensemble size $K\in\{1,2,\dots\}$. Let $\Theta=\R^D$ denote the single-neuron
state space and let $(\mathcal X\times\mathcal Y,\mathsf P)$ be the data space.

For each head $\alpha\in\{1,\dots,K\}$, let
\[
  \Psi_\alpha:\mathcal X\times\Theta\to\R
\]
be a feature map. Given a neuron law $\rho\in\PP_1(\Theta)$, define the head
predictor
\begin{equation}
  f_\alpha(x;\rho)
  :=
  \int_\Theta \Psi_\alpha(x;\theta)\,\rho(\dd\theta).
  \label{eq:output}
\end{equation}
The head-averaged population risk is
\begin{equation}
  \mathcal R(\rho)
  :=
  \E_{(x,y)\sim\mathsf P}
  \Bigl[
    \frac1K\sum_{\alpha=1}^K \ell(f_\alpha(x;\rho),y)
  \Bigr].
  \label{eq:risk}
\end{equation}

\begin{definition}[Mean-field drift]
Assume $\ell$ is differentiable in its first argument. Define
\begin{equation}
  B(\theta;\rho)
  :=
  -\E_{(x,y)\sim\mathsf P}
  \Bigl[
    \frac1K\sum_{\alpha=1}^K
    \partial_1\ell(f_\alpha(x;\rho),y)\,
    \nabla_\theta \Psi_\alpha(x;\theta)
  \Bigr].
  \label{eq:drift}
\end{equation}
The no-input fixed-$K$ mean-field equation is
\begin{equation}
  \partial_t \rho_t
  +
  \nabla_\theta\cdot\bigl(\rho_t B(\theta;\rho_t)\bigr)
  =
  0.
  \label{eq:pde}
\end{equation}
\end{definition}

\begin{definition}[Characteristic solutions]
Fix $T>0$ and $\rho_0\in\PP_1(\Theta)$. A curve
\[
  (\rho_t)_{t\in[0,T]}\in C([0,T],\PP_1(\Theta))
\]
is a characteristic solution of \cref{eq:pde} if there exists a flow
\[
  \Phi:[0,T]\times\Theta\to\Theta
\]
such that, for $\rho_0$-almost every $\theta$,
\[
  \dot\Phi_t(\theta)=B(\Phi_t(\theta);\rho_t),
  \qquad
  \Phi_0(\theta)=\theta,
  \qquad
  \rho_t=(\Phi_t)_\#\rho_0.
\]
\end{definition}

\subsection{Structural assumptions}

Let $S_K$ be the permutation group on $\{1,\dots,K\}$. Assume that for each
$\pi\in S_K$ there is a linear isometry $\Pi_\pi^\Theta:\Theta\to\Theta$
representing the permutation of head-labelled coordinates inside the neuron
state.

\begin{assumption}[Bounded-Lipschitz structure and permutation equivariance]
\label{ass:main}
There exist finite constants
\[
  B_\Psi,\ B_{\nabla\Psi},\ L_\Psi,\ L_{\nabla\Psi},\ L_\ell,\ C_\ell
\]
such that the following hold uniformly over all $\alpha\in\{1,\dots,K\}$ and
$x\in\mathcal X$.
\begin{enumerate}[label=(\roman*),leftmargin=2em]
  \item The maps $\Psi_\alpha(x;\cdot)$ and $\nabla_\theta\Psi_\alpha(x;\cdot)$
  are globally bounded by $B_\Psi$ and $B_{\nabla\Psi}$ and globally Lipschitz
  in $\theta$ with constants $L_\Psi$ and $L_{\nabla\Psi}$.
  \item $\partial_1\ell(\cdot,y)$ is globally Lipschitz with constant $L_\ell$
  and uniformly bounded by $C_\ell$ for all $y\in\mathcal Y$.
  \item The feature maps are permutation-equivariant:
  \begin{equation}
    \Psi_\alpha(x;\Pi_\pi^\Theta\theta)
    =
    \Psi_{\pi^{-1}(\alpha)}(x;\theta)
    \qquad
    \forall \pi\in S_K.
    \label{eq:equiv}
  \end{equation}
\end{enumerate}
\end{assumption}

\begin{remark}[Concrete bias-free two-layer BatchEnsemble model]
\label{rem:concrete}
Fix an input dimension $d_x$ and a hidden width $m$. Let
\[
  \theta_j=(a_j,w_j,r_{1,j},\dots,r_{K,j})
  \in
  \Theta_{\mathrm{BE}}
  :=
  \R\times\R^{d_x}\times\R^K.
\]
The finite-width network is
\begin{equation}
  f_\alpha^m(x)
  :=
  \frac1m\sum_{j=1}^m
  a_j\,\sigma\!\bigl(r_{\alpha,j}\,w_j^\top x\bigr),
  \qquad
  \alpha\in\{1,\dots,K\},
  \label{eq:concrete-model}
\end{equation}
with empirical neuron law
\[
  \rho^m:=\frac1m\sum_{j=1}^m\delta_{\theta_j}.
\]
Then \eqref{eq:concrete-model} is of the abstract form \eqref{eq:output} with
\begin{equation}
  \Psi_\alpha^{\mathrm{BE}}
  \bigl(x;(a,w,r_1,\dots,r_K)\bigr)
  :=
  a\,\sigma(r_\alpha w^\top x).
  \label{eq:psi-concrete}
\end{equation}
After composing the raw coordinates with smooth clipping maps whose images and
first derivatives are globally bounded and permutation-equivariant,
\eqref{eq:psi-concrete} fits \cref{ass:main}. This note therefore captures the
rigorous core of the no-input-modulation width limit.
\end{remark}

\section{Deterministic Mean-Field Theory}

\begin{proposition}[Output and drift Lipschitz bounds]
\label{prop:lipschitz}
Assume \cref{ass:main}. Then for every $x\in\mathcal X$ and every $\alpha$,
\begin{equation}
  |f_\alpha(x;\rho)-f_\alpha(x;\mu)|
  \le
  L_\Psi\,\W(\rho,\mu).
  \label{eq:out-lip}
\end{equation}
Moreover, there exists a finite constant $L_B$ depending only on the constants
in \cref{ass:main} such that
\begin{equation}
  \|B(\theta;\rho)-B(\eta;\mu)\|
  \le
  L_B\bigl(\|\theta-\eta\|+\W(\rho,\mu)\bigr).
  \label{eq:B-lip}
\end{equation}
\end{proposition}

\begin{proof}
The output bound \eqref{eq:out-lip} follows immediately from the dual definition
of $\W$ and the $L_\Psi$-Lipschitz property of $\Psi_\alpha(x;\cdot)$. For the
drift, write
\[
  B(\theta;\rho)-B(\eta;\mu)=I_1+I_2,
\]
where
\begin{align*}
  I_1
  &:=
  -\E\Bigl[
    \frac1K\sum_{\alpha=1}^K
    \bigl(
      \partial_1\ell(f_\alpha(x;\rho),y)-\partial_1\ell(f_\alpha(x;\mu),y)
    \bigr)\nabla_\theta\Psi_\alpha(x;\theta)
  \Bigr],\\
  I_2
  &:=
  -\E\Bigl[
    \frac1K\sum_{\alpha=1}^K
    \partial_1\ell(f_\alpha(x;\mu),y)\,
    \bigl(
      \nabla_\theta\Psi_\alpha(x;\theta)-\nabla_\theta\Psi_\alpha(x;\eta)
    \bigr)
  \Bigr].
\end{align*}
By \eqref{eq:out-lip}, the Lipschitz property of $\partial_1\ell$, and the
bound on $\nabla_\theta\Psi_\alpha$, we get
\[
  \|I_1\|\le L_\ell B_{\nabla\Psi}L_\Psi\,\W(\rho,\mu).
\]
By the uniform bound on $\partial_1\ell$ and the Lipschitz property of
$\nabla_\theta\Psi_\alpha$,
\[
  \|I_2\|\le C_\ell L_{\nabla\Psi}\,\|\theta-\eta\|.
\]
Adding the two bounds gives \eqref{eq:B-lip}.
\end{proof}

\begin{theorem}[Deterministic well-posedness, stability, and particle convergence]
\label{thm:deterministic}
Assume \cref{ass:main}, fix $T>0$, and let $\rho_0\in\PP_1(\Theta)$.

\begin{enumerate}[label=(\roman*),leftmargin=2em]
  \item There exists a unique characteristic solution
  \[
    (\rho_t)_{t\in[0,T]}\in C([0,T],\PP_1(\Theta))
  \]
  of \cref{eq:pde}.
  \item There exists a constant $C_T<\infty$ such that for any two initial laws
  $\rho_0,\mu_0\in\PP_1(\Theta)$ with corresponding solutions $(\rho_t)$ and
  $(\mu_t)$,
  \begin{equation}
    \sup_{t\in[0,T]}\W(\rho_t,\mu_t)
    \le
    C_T\,\W(\rho_0,\mu_0).
    \label{eq:det-stability}
  \end{equation}
  \item Let deterministic particles satisfy
  \[
    \dot\theta_{j,t}^m=B(\theta_{j,t}^m;\rho_t^m),
    \qquad
    \rho_t^m=\frac1m\sum_{j=1}^m\delta_{\theta_{j,t}^m},
    \qquad
    \theta_{j,0}^m\in\Theta.
  \]
  If $\rho_0^m:=m^{-1}\sum_{j=1}^m\delta_{\theta_{j,0}^m}$ converges to $\rho_0$
  in $\W$, then
  \begin{equation}
    \sup_{t\in[0,T]}\W(\rho_t^m,\rho_t)
    \le
    C_T\,\W(\rho_0^m,\rho_0)
    \xrightarrow[m\to\infty]{}0.
    \label{eq:particle-conv}
  \end{equation}
  In particular, for every $\alpha$ and $x$,
  \[
    \sup_{t\in[0,T]}
    |f_\alpha(x;\rho_t^m)-f_\alpha(x;\rho_t)|
    \le
    L_\Psi\sup_{t\in[0,T]}\W(\rho_t^m,\rho_t)
    \to 0.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
By \cref{prop:lipschitz}, the map $\theta\mapsto B(\theta;\rho)$ is globally
Lipschitz uniformly on bounded $\W$-balls, and the map $\rho\mapsto B(\cdot;\rho)$
is Lipschitz in $\W$. Standard fixed-point arguments on the flow map therefore
yield existence and uniqueness of a characteristic solution on any finite time
interval.

For stability, let $\Phi_t$ and $\Psi_t$ denote the characteristic flows driven
by $(\rho_t)$ and $(\mu_t)$, respectively. Start from an optimal coupling
$\lambda_0$ of $\rho_0$ and $\mu_0$, and transport it by
$(\Phi_t,\Psi_t)$ to obtain a coupling $\lambda_t$ of $\rho_t$ and $\mu_t$.
Then
\begin{align*}
  \frac{\dd}{\dd t}
  \int \|\Phi_t(\theta)-\Psi_t(\eta)\|\,\lambda_0(\dd\theta,\dd\eta)
  &\le
  \int
    \|B(\Phi_t(\theta);\rho_t)-B(\Psi_t(\eta);\mu_t)\|
    \,\lambda_0(\dd\theta,\dd\eta) \\
  &\le
  L_B
  \int
    \|\Phi_t(\theta)-\Psi_t(\eta)\|
    \,\lambda_0(\dd\theta,\dd\eta)
  +
  L_B\,\W(\rho_t,\mu_t).
\end{align*}
Since $\W(\rho_t,\mu_t)$ is bounded by the same integral, Gronwall yields
\eqref{eq:det-stability}.

For the particle system, observe that $\rho_t^m$ is itself a characteristic
solution of \cref{eq:pde} with initial law $\rho_0^m$, because the empirical
measure is transported by the particle flow. Applying \eqref{eq:det-stability}
with $\mu_t=\rho_t^m$ yields \eqref{eq:particle-conv}. The predictor convergence
then follows from \eqref{eq:out-lip}.
\end{proof}

\section{Random Initialization and Symmetry}

\begin{corollary}[Random initialization with deterministic law]
\label{cor:random}
Assume \cref{ass:main}, fix $T>0$, and let $\rho_0\in\PP_1(\Theta)$ be
deterministic. Suppose the random empirical initial laws $\rho_0^m$ satisfy
\[
  \W(\rho_0^m,\rho_0)\xrightarrow[m\to\infty]{}0
\]
in probability, almost surely, or in $L^1$. Let $\rho_t^m$ be the corresponding
particle empirical law and let $(\rho_t)$ be the deterministic solution of
\cref{eq:pde} started from $\rho_0$. Then the same mode of convergence holds
uniformly on $[0,T]$:
\[
  \sup_{t\in[0,T]}\W(\rho_t^m,\rho_t)\to 0.
\]
In particular, the infinite-width predictor vector
\[
  \bigl(f_1(\cdot;\rho_t),\dots,f_K(\cdot;\rho_t)\bigr)
\]
is deterministic.
\end{corollary}

\begin{proof}
This is immediate from \eqref{eq:particle-conv}.
\end{proof}

\begin{proposition}[Permutation symmetry is preserved]
\label{prop:symmetry}
Assume \cref{ass:main}. If $\rho_0\in\PP_1(\Theta)$ is invariant under every
$\Pi_\pi^\Theta$, i.e.
\[
  (\Pi_\pi^\Theta)_\#\rho_0=\rho_0
  \qquad
  \forall \pi\in S_K,
\]
then the unique characteristic solution of \cref{eq:pde} satisfies
\[
  (\Pi_\pi^\Theta)_\#\rho_t=\rho_t
  \qquad
  \forall t\ge 0,\ \forall \pi\in S_K.
\]
Consequently,
\begin{equation}
  f_1(x;\rho_t)=\cdots=f_K(x;\rho_t)
  \qquad
  \forall x\in\mathcal X,\ \forall t\ge 0.
  \label{eq:collapse}
\end{equation}
\end{proposition}

\begin{proof}
Define $\tilde\rho_t:=(\Pi_\pi^\Theta)_\#\rho_t$. Using \eqref{eq:equiv}, a
change of variables gives
\begin{align*}
  f_\alpha(x;\tilde\rho_t)
  &=
  \int \Psi_\alpha(x;\theta)\,\tilde\rho_t(\dd\theta) \\
  &=
  \int \Psi_\alpha(x;\Pi_\pi^\Theta\eta)\,\rho_t(\dd\eta) \\
  &=
  \int \Psi_{\pi^{-1}(\alpha)}(x;\eta)\,\rho_t(\dd\eta) \\
  &=
  f_{\pi^{-1}(\alpha)}(x;\rho_t).
\end{align*}
Applying the same identity to the drift shows
\[
  B(\Pi_\pi^\Theta\theta;\tilde\rho_t)
  =
  \Pi_\pi^\Theta B(\theta;\rho_t).
\]
Hence, if $\Phi_t$ is the characteristic flow of $\rho_t$, then
$\Pi_\pi^\Theta\circ\Phi_t\circ(\Pi_\pi^\Theta)^{-1}$ is the characteristic flow
of $\tilde\rho_t$. Since $\rho_0$ is invariant, both $\rho_t$ and $\tilde\rho_t$
solve \cref{eq:pde} with the same initial law, so uniqueness gives
$\tilde\rho_t=\rho_t$.

Finally, for any $\alpha,\beta$ choose a permutation $\pi$ with
$\pi^{-1}(\alpha)=\beta$. Using invariance of $\rho_t$ and \eqref{eq:equiv},
\[
  f_\alpha(x;\rho_t)
  =
  \int \Psi_\alpha(x;\theta)\,\rho_t(\dd\theta)
  =
  \int \Psi_\alpha(x;\Pi_\pi^\Theta\theta)\,\rho_t(\dd\theta)
  =
  \int \Psi_{\pi^{-1}(\alpha)}(x;\theta)\,\rho_t(\dd\theta)
  =
  f_\beta(x;\rho_t),
\]
which proves \eqref{eq:collapse}.
\end{proof}

\begin{corollary}[Standard symmetric initialization collapses at leading order]
\label{cor:collapse}
Assume the setting of \cref{cor:random}, and in addition assume that the limit
law $\rho_0$ is permutation-invariant under all $\Pi_\pi^\Theta$. Then the
deterministic infinite-width limit satisfies \eqref{eq:collapse}. In particular,
for any fixed head indices $\alpha,\beta$ and any $x$,
\[
  f_\alpha(x;\rho_t)-f_\beta(x;\rho_t)=0
  \qquad
  \forall t\ge 0.
\]
\end{corollary}

\section{Discussion}

The no-input-modulation regime has a qualitatively simpler width limit than the
setting with head-global variables.

\paragraph{1. The correct state variable is a single neuron law.}
Because all head-specific variables are already coordinates of the single-neuron
state, fixed-$K$ and $m\to\infty$ lead to a single transport PDE. No additional
ODE for head-global variables is needed.

\paragraph{2. Standard width limits are deterministic.}
If the empirical initialization converges to a deterministic law, then the
infinite-width predictor vector is deterministic. There is no surviving
order-one random environment analogous to the case with learned input-side
modulations.

\paragraph{3. Symmetry now forces leading-order collapse.}
If the initialization law is permutation-symmetric in the head coordinates, then
the mean-field flow preserves that symmetry, and all head predictors coincide in
the limit. In this regime, any observed diversity between heads is therefore
subleading: it must come from finite-width fluctuations, deliberately asymmetric
initialization, or a different asymptotic regime.

\paragraph{4. This is the natural precursor to the richer models.}
The note cleanly separates what width alone can do from what requires additional
head-global structure. Reintroducing input-side modulation $s_\alpha$ turns the
single-PDE picture into a PDE-ODE system; taking $K\to\infty$ turns it into a
different mean-field problem again.
